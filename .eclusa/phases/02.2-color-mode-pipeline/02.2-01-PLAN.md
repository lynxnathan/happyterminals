/// Terminal color capability tier.
/// `TrueColor` = 24-bit RGB; `Palette256` = xterm 8-bit indexed;
/// `Ansi16` = 4-bit named system colors; `Mono` = no color, modifiers preserved.
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum ColorMode {
    TrueColor,
    Palette256,
    Ansi16,
    Mono,
}

/// Minimal env-read abstraction so tests can inject fake env without
/// touching `std::env::set_var` (Pitfall 4: parallel test env-leak).
pub trait EnvProvider {
    fn var(&self, key: &str) -> Option<String>;
}

/// Production impl — reads `std::env::var` and returns `None` on any error
/// (VarError::NotPresent OR VarError::NotUnicode treated as unset per RESEARCH).
pub struct RealEnv;
impl EnvProvider for RealEnv { /* ... */ }

/// Pure cascade: `NO_COLOR` → `override_mode` → `$COLORTERM` → `$TERM`.
/// Called exactly once at `run()` entry (Plan 02 wires this).
pub fn detect_color_mode<E: EnvProvider>(
    override_mode: Option<ColorMode>,
    env: &E,
) -> ColorMode;

/// Thin wrapper: `detect_color_mode(override_mode, &RealEnv)`.
pub fn detect_color_mode_from_real_env(override_mode: Option<ColorMode>) -> ColorMode;

/// Flush-time transform — mutates the buffer in place.
/// TrueColor is a no-op fast path.
pub fn downsample(buffer: &mut ratatui_core::buffer::Buffer, mode: ColorMode);
```

    //! xterm 256-color palette + compile-time 256→16 quantization LUT.
    //!
    //! Source: Wikipedia ANSI escape code spec (HIGH confidence per RESEARCH).
    //! All arrays are built in `const fn`; MSRV 1.88 covers all required
    //! const features (verified in RESEARCH §Pitfall 8).

    /// sRGB channel levels for the 6×6×6 cube (xterm standard).
    const CUBE_LEVELS: [u8; 6] = [0, 95, 135, 175, 215, 255];

    /// xterm system-16 palette (xterm variant of the ANSI 16 named colors).
    /// Indexed by the low nibble of the 8-bit color code.
    pub(crate) const SYSTEM_16: [(u8, u8, u8); 16] = [
        (0x00, 0x00, 0x00), (0x80, 0x00, 0x00), (0x00, 0x80, 0x00), (0x80, 0x80, 0x00),
        (0x00, 0x00, 0x80), (0x80, 0x00, 0x80), (0x00, 0x80, 0x80), (0xC0, 0xC0, 0xC0),
        (0x80, 0x80, 0x80), (0xFF, 0x00, 0x00), (0x00, 0xFF, 0x00), (0xFF, 0xFF, 0x00),
        (0x00, 0x00, 0xFF), (0xFF, 0x00, 0xFF), (0x00, 0xFF, 0xFF), (0xFF, 0xFF, 0xFF),
    ];

    /// Standard xterm 256-color palette.
    /// - 0..16:  ANSI system colors (xterm variant)
    /// - 16..232: 6×6×6 RGB cube, ordered as `16 + 36*r + 6*g + b`
    ///   with channel levels `[0, 95, 135, 175, 215, 255]`
    /// - 232..256: 24 greyscale steps, level = `8 + 10 * (i - 232)`
    pub const PALETTE_256: [(u8, u8, u8); 256] = build_palette();

    /// Compile-time 256→16 quantization LUT (sRGB-nearest over SYSTEM_16).
    pub const PALETTE_16_LUT: [u8; 256] = build_16_lut();

    const fn build_palette() -> [(u8, u8, u8); 256] {
        let mut out = [(0u8, 0u8, 0u8); 256];
        let mut i = 0;
        while i < 16 { out[i] = SYSTEM_16[i]; i += 1; }
        let mut r = 0;
        while r < 6 {
            let mut g = 0;
            while g < 6 {
                let mut b = 0;
                while b < 6 {
                    out[16 + 36*r + 6*g + b] =
                        (CUBE_LEVELS[r], CUBE_LEVELS[g], CUBE_LEVELS[b]);
                    b += 1;
                }
                g += 1;
            }
            r += 1;
        }
        let mut k = 0;
        while k < 24 {
            let v = 8 + 10 * (k as u8);
            out[232 + k] = (v, v, v);
            k += 1;
        }
        out
    }

    const fn build_16_lut() -> [u8; 256] {
        let mut out = [0u8; 256];
        let mut i = 0;
        while i < 256 {
            let target = PALETTE_256[i];
            let mut best_j = 0u8;
            let mut best_d = u32::MAX;
            let mut j = 0;
            while j < 16 {
                let d = sq_dist(target, SYSTEM_16[j]);
                if d < best_d { best_d = d; best_j = j as u8; }
                j += 1;
            }
            out[i] = best_j;
            i += 1;
        }
        out
    }

    /// Squared Euclidean distance in sRGB. Ordering-preserving — no sqrt.
    #[inline]
    pub(crate) const fn sq_dist(a: (u8, u8, u8), b: (u8, u8, u8)) -> u32 {
        let dr = a.0 as i32 - b.0 as i32;
        let dg = a.1 as i32 - b.1 as i32;
        let db = a.2 as i32 - b.2 as i32;
        (dr * dr + dg * dg + db * db) as u32
    }

    /// Linear scan over `PALETTE_256`. Returns the first index minimizing
    /// squared sRGB distance to `rgb`. Ties resolve to the lower index.
    #[must_use]
    pub fn nearest_256(rgb: (u8, u8, u8)) -> u8 {
        let mut best_i = 0u8;
        let mut best_d = u32::MAX;
        let mut i = 0u16;
        while i < 256 {
            let d = sq_dist(rgb, PALETTE_256[i as usize]);
            if d < best_d { best_d = d; best_i = i as u8; }
            i += 1;
        }
        best_i
    }

    /// across parallel test threads — see RESEARCH §Pitfall 4).
    pub trait EnvProvider {
        /// Returns the env var value, or `None` if unset OR non-unicode.
        fn var(&self, key: &str) -> Option<String>;
    }

    /// Production impl — reads via `std::env::var`. `VarError::NotUnicode`
    /// is treated as unset (pathological case; non-issue on Linux/Windows
    /// per RESEARCH §Pattern 1 note).
    pub struct RealEnv;

    impl EnvProvider for RealEnv {
        fn var(&self, key: &str) -> Option<String> {
            std::env::var(key).ok()
        }
    }

    /// Follows the cascade (highest priority first):
    /// 1. `NO_COLOR` present AND non-empty → `Mono`
    /// 2. `override_mode` present → forced
    /// 3. `$COLORTERM` equals `truecolor` or `24bit` (case-insensitive) → `TrueColor`
    /// 4. `$TERM` inspection:
    ///    - `"dumb"` → `Mono`
    ///    - contains `"256color"` → `Palette256`
    ///    - `"xterm-kitty" | "alacritty"` → `TrueColor`
    ///    - anything else → `Ansi16`
    ///    - unset → `Mono`
    #[must_use]
    pub fn detect_color_mode<E: EnvProvider>(
        override_mode: Option<ColorMode>,
        env: &E,
    ) -> ColorMode {
        // 1. NO_COLOR (highest priority; beats override per no-color.org)
        if let Some(v) = env.var("NO_COLOR") {
            if !v.is_empty() { return ColorMode::Mono; }
        }
        // 2. Programmatic override
        if let Some(mode) = override_mode { return mode; }
        // 3. COLORTERM truecolor signal
        if let Some(v) = env.var("COLORTERM") {
            let lower = v.to_ascii_lowercase();
            if lower == "truecolor" || lower == "24bit" {
                return ColorMode::TrueColor;
            }
        }
        // 4. TERM fallback
        match env.var("TERM").as_deref() {
            Some("dumb") => ColorMode::Mono,
            Some(t) if t.contains("256color") => ColorMode::Palette256,
            Some("xterm-kitty") | Some("alacritty") => ColorMode::TrueColor,
            Some(_) => ColorMode::Ansi16,
            None => ColorMode::Mono,
        }
    }

    /// Convenience wrapper over [`detect_color_mode`] using [`RealEnv`].
    #[must_use]
    pub fn detect_color_mode_from_real_env(override_mode: Option<ColorMode>) -> ColorMode {
        detect_color_mode(override_mode, &RealEnv)
    }
    ```

    Step 3 — `downsample` + `map_color` + `named_from_u8`:

    ```rust
    use ratatui_core::buffer::Buffer;
    use ratatui_core::layout::Position;
    use ratatui_core::style::Color;

    /// Applies the color-mode transform to every cell in `buffer`.
    /// `TrueColor` is a no-op fast path. Modifiers (bold/italic/underline/reverse)
    /// are preserved across all modes.
    pub fn downsample(buffer: &mut Buffer, mode: ColorMode) {
        if matches!(mode, ColorMode::TrueColor) { return; }
        let area = buffer.area();
        for y in area.top()..area.bottom() {
            for x in area.left()..area.right() {
                let cell = buffer.cell_mut(Position::new(x, y));
                if let Some(cell) = cell {
                    cell.set_fg(map_color(cell.fg, mode));
                    cell.set_bg(map_color(cell.bg, mode));
                    // modifier preserved — no touch
                }
            }
        }
    }

    fn map_color(c: Color, mode: ColorMode) -> Color {
        match mode {
            ColorMode::TrueColor => c,
            ColorMode::Mono => match c {
                Color::Reset => Color::Reset,
                _            => Color::Reset,  // Pitfall 7: NOT Color::Black
            },
            ColorMode::Palette256 => match c {
                Color::Rgb(r, g, b) => Color::Indexed(nearest_256((r, g, b))),
                other               => other,
            },
            ColorMode::Ansi16 => match c {
                Color::Rgb(r, g, b) => {
                    let idx16 = PALETTE_16_LUT[nearest_256((r, g, b)) as usize];
                    named_from_u8(idx16)
                }
                Color::Indexed(i) => named_from_u8(PALETTE_16_LUT[i as usize]),
                other             => other,
            },
        }
    }

    fn named_from_u8(i: u8) -> Color {
        match i {
            0  => Color::Black,        1 => Color::Red,          2 => Color::Green,
            3  => Color::Yellow,       4 => Color::Blue,         5 => Color::Magenta,
            6  => Color::Cyan,         7 => Color::Gray,         8 => Color::DarkGray,
            9  => Color::LightRed,    10 => Color::LightGreen,  11 => Color::LightYellow,
            12 => Color::LightBlue,  13 => Color::LightMagenta, 14 => Color::LightCyan,
            _  => Color::White,
        }
    }
    ```
