    ```rust
    //! `TerminalGuard` — RAII wrapper... (existing doc unchanged)

    use std::io::{self, Stdout, Write};

    use crossterm::{
        cursor,
        event::{DisableMouseCapture, EnableMouseCapture},
        execute,
        terminal::{self, EnterAlternateScreen, LeaveAlternateScreen},
    };

    use crate::color::ColorMode;

    /// RAII guard that owns the terminal's raw-mode / alternate-screen state
    /// and caches the resolved [`ColorMode`] for the lifetime of the run.
    pub struct TerminalGuard {
        stdout: Stdout,
        /// Cached color-mode resolved once at acquire-time. Read by the flush
        /// path to drive the downsample pass; never mutated after construction.
        pub color_mode: ColorMode,
        #[cfg(test)]
        /// Test-only flag: when `true`, skip raw-mode/alt-screen restore on drop.
        /// Used by `with_mode_uninitialized_for_test`.
        uninitialized: bool,
    }

    impl TerminalGuard {
        /// Enters raw mode, alt screen, mouse capture; caches `ColorMode::TrueColor`.
        /// Back-compat shim — prefer [`acquire_with_color_mode`] for explicit mode.
        pub fn acquire() -> io::Result<Self> {
            Self::acquire_with_color_mode(ColorMode::TrueColor)
        }

        /// Enters raw mode, alt screen, mouse capture; caches the given `ColorMode`.
        pub fn acquire_with_color_mode(color_mode: ColorMode) -> io::Result<Self> {
            let mut stdout = io::stdout();
            terminal::enable_raw_mode()?;
            execute!(
                stdout,
                EnterAlternateScreen,
                EnableMouseCapture,
                cursor::Hide
            )?;
            Ok(Self {
                stdout,
                color_mode,
                #[cfg(test)]
                uninitialized: false,
            })
        }

        /// Test-only: construct a guard WITHOUT entering raw mode / alt screen.
        /// Drop is a no-op for this variant. Used to verify field caching.
        #[cfg(test)]
        pub(crate) fn with_mode_uninitialized_for_test(color_mode: ColorMode) -> Self {
            Self {
                stdout: io::stdout(),
                color_mode,
                uninitialized: true,
            }
        }

        pub fn restore(stdout: &mut Stdout) {
            let _ = execute!(stdout, cursor::Show, DisableMouseCapture, LeaveAlternateScreen);
            let _ = terminal::disable_raw_mode();
            let _ = stdout.write_all(b"\x1b[0m");
            let _ = stdout.flush();
        }
    }

    impl Drop for TerminalGuard {
        fn drop(&mut self) {
            #[cfg(test)]
            { if self.uninitialized { return; } }
            Self::restore(&mut self.stdout);
        }
    }

    pub fn install_panic_hook() { /* unchanged */ }

    #[cfg(test)]
    mod tests { /* existing + test_guard_caches_color_mode */ }
    ```
